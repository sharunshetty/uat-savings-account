CREATE SEQUENCE SBREQ_MASTER_SEQ
START WITH 1000001
MAXVALUE   9999999
INCREMENT BY 1
CYCLE
CACHE 50;

-- SELECT 'N' || DATE_FORMAT(SYSDATE(),'%Y%m%d') || NEXTVAL(SBREQ_MASTER_SEQ) FROM DUAL

CREATE TABLE SB_REQUESTS (
  SBR_REF_NUM VARCHAR(30) NOT NULL,
  SBR_CUST_NAME VARCHAR(120),
  SBR_APP_STATUS CHAR(3),
  SBR_CUST_MOBILE VARCHAR(15),
  SBR_CUST_EMAIL VARCHAR(120),
  SBR_CUST_DOB DATE,
  SBR_MINOR_FLAG CHAR(1),
  SBR_CUST_GENDER CHAR(1),
  SBR_REFERRAL_CODE VARCHAR(120),
  SBR_AADHAAR_NUM VARCHAR(20),
  SBR_AADHAAR_CR_ON DATETIME,
  SBR_PAN_NUM VARCHAR(20),
  SBR_PAN_CR_ON DATETIME,
  CR_BY VARCHAR(30),
  CR_ON DATETIME,
  MO_BY VARCHAR(30),
  MO_ON DATETIME,
  AU_BY VARCHAR(30),
  AU_ON DATETIME,
  TBA_KEY VARCHAR(25),
  PRIMARY KEY (SBR_REF_NUM)
);
